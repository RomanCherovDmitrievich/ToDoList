473 return;

476 Dragboard db = row.startDragAndDrop(TransferMode.MOVE);
477 ClipboardContent content = new ClipboardContent();
478 content.putString(draggedTask.getId());
479 db.setContent(content);
480 event.consume();
481 });
482
483 row.setOnDragOver(event -> {
484 if (draggedTask == null || row.isEmpty()) {
485 return;
486 }
487 if (row.getItem() != draggedTask) {
488 event.acceptTransferModes(TransferMode.MOVE);
489 }
490 event.consume();
491 });
492
493 row.setOnDragDropped(event -> {
494 if (draggedTask == null || row.isEmpty()) {
495 return;
496 }
497 TaskViewModel target = row.getItem();
498 int draggedIdx = taskList.indexOf(draggedTask);
499 int targetIdx = taskList.indexOf(target);
500 if (draggedIdx < 0 || targetIdx < 0) {
501 return;
502 }
503 if (draggedIdx != targetIdx) {
504 taskList.remove(draggedIdx);
505 int insertIndex = targetIdx;
506 if (insertIndex > taskList.size()) {
507 insertIndex = taskList.size();
508 }
509 taskList.add(insertIndex, draggedTask);
510 persistOrder();
511 }
512 draggedTask = null;
513 event.setDropCompleted(true);
514 refreshAllViews();
515 });
516
517 row.setOnDragDone(event -> draggedTask = null);
518
519 ContextMenu menu = buildContextMenu(row);
520 row.contextMenuProperty().bind(javafx.beans.binding.Bindings
521 .when(row.emptyProperty())
522 .then((ContextMenu) null)
523 .otherwise(menu));
524
525 return row;
526 });

682 private void applySortMode() {
683 if (sortMode == SortMode.MANUAL) {
684 Comparator<TaskViewModel> comparator = Comparator
685 .comparingInt((TaskViewModel vm) -> vm.getTask().getSortIndex())
686 .thenComparing(TaskViewModel::getEndTime);
687 FXCollections.sort(taskList, comparator);
688 return;
689 }
690
691 if (sortMode == SortMode.DATE) {
692 Comparator<TaskViewModel> comparator = Comparator.comparing(TaskViewModel::getEndTime);
693 if (!sortByDateAscending) {
694 comparator = comparator.reversed();
695 }
696 FXCollections.sort(taskList, comparator);
697 sortByDateButton.setText(sortByDateAscending ? "По дате ↑" : "По дате ↓");
698 return;
699 }
700
701 Comparator<TaskViewModel> comparator = Comparator
702 .comparing((TaskViewModel vm) -> priorityRank(vm.getPriority()))
703 .thenComparing(TaskViewModel::getEndTime);
704 if (sortByPriorityAscending) {
705 comparator = comparator.reversed();
706 }
707 FXCollections.sort(taskList, comparator);
708 sortByPriorityButton.setText(sortByPriorityAscending ? "По приоритету ↓" : "По приоритету ↑");
709 }
