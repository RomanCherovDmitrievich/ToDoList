527 public String getTimeRemaining() {
528 LocalDateTime now = LocalDateTime.now();
529 LocalDateTime deadline = endTime.get();
530
531 if (now.isAfter(deadline)) {
532 // Задача просрочена
533 long days = now.toLocalDate().toEpochDay() - deadline.toLocalDate().toEpochDay();
534 if (days == 0) {
535 return "Просрочено сегодня";
536 } else if (days == 1) {
537 return "Просрочено вчера";
538 } else {
539 return "Просрочено " + days + " дн. назад";
540 }
541 } else {
542 // Задача еще не просрочена
543 long days = deadline.toLocalDate().toEpochDay() - now.toLocalDate().toEpochDay();
544 if (days == 0) {
545 return "Сегодня";
546 } else if (days == 1) {
547 return "Завтра";
548 } else {
549 return "Через " + days + " дн.";
550 }
551 }
552 }

565 public String getStatusText() {
566 if (completed.get()) {
567 return "Выполнено";
568 } else {
569 // Проверяем актуальную просроченность
570 boolean isActuallyOverdue = LocalDateTime.now().isAfter(endTime.get());
571 if (isActuallyOverdue) {
572 // Обновляем свойство overdue
573 overdue.set(true);
574 return "Просрочено";
575 } else {
576 return "Активно";
577 }
578 }
