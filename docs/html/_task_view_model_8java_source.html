513 public String getTimeRemaining() {
514 LocalDateTime now = LocalDateTime.now();
515 LocalDateTime deadline = endTime.get();
516
517 if (now.isAfter(deadline)) {
518 // Задача просрочена
519 long days = now.toLocalDate().toEpochDay() - deadline.toLocalDate().toEpochDay();
520 if (days == 0) {
521 return "Просрочено сегодня";
522 } else if (days == 1) {
523 return "Просрочено вчера";
524 } else {
525 return "Просрочено " + days + " дн. назад";
526 }
527 } else {
528 // Задача еще не просрочена
529 long days = deadline.toLocalDate().toEpochDay() - now.toLocalDate().toEpochDay();
530 if (days == 0) {
531 return "Сегодня";
532 } else if (days == 1) {
533 return "Завтра";
534 } else {
535 return "Через " + days + " дн.";
536 }
537 }
538 }
