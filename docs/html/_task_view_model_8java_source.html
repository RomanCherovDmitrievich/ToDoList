518 public String getTimeRemaining() {
519 LocalDateTime now = LocalDateTime.now();
520 LocalDateTime deadline = endTime.get();
521
522 if (now.isAfter(deadline)) {
523 // Задача просрочена
524 long days = now.toLocalDate().toEpochDay() - deadline.toLocalDate().toEpochDay();
525 if (days == 0) {
526 return "Просрочено сегодня";
527 } else if (days == 1) {
528 return "Просрочено вчера";
529 } else {
530 return "Просрочено " + days + " дн. назад";
531 }
532 } else {
533 // Задача еще не просрочена
534 long days = deadline.toLocalDate().toEpochDay() - now.toLocalDate().toEpochDay();
535 if (days == 0) {
536 return "Сегодня";
537 } else if (days == 1) {
538 return "Завтра";
539 } else {
540 return "Через " + days + " дн.";
541 }
542 }
543 }

556 public String getStatusText() {
557 if (completed.get()) {
558 return "Выполнено";
559 } else {
560 // Проверяем актуальную просроченность
561 boolean isActuallyOverdue = LocalDateTime.now().isAfter(endTime.get());
562 if (isActuallyOverdue) {
563 // Обновляем свойство overdue
564 overdue.set(true);
565 return "Просрочено";
566 } else {
567 return "Активно";
568 }
569 }
