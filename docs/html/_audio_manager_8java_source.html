79 public synchronized void refreshPlaylist() {
80 playlist.clear();
81
82 Set<String> unique = new HashSet<>();
83
84 // 1) Внешняя папка (приоритетно для пользовательской музыки)
85 for (Path dir : resolveExternalAudioDirs()) {
86 if (!Files.isDirectory(dir)) {
87 continue;
88 }
89 try {
90 Files.list(dir)
91 .filter(Files::isRegularFile)
92 .filter(path -> isAudioFile(path.getFileName().toString()))
93 .sorted(Comparator.comparing(path -> path.getFileName().toString().toLowerCase()))
94 .forEach(path -> {
95 URI uri = path.toUri();
96 if (unique.add(uri.toString())) {
97 playlist.add(uri);
98 }
99 });
100 } catch (Exception e) {
101 System.err.println("Audio scan error (" + dir + "): " + e.getMessage());
102 }
103 }
104
105 // 2) Встроенные fallback-треки
106 addBundledTrack("/resources/audio/startup.mp3", unique);
107 addBundledTrack("/resources/audio/startup2.mp3", unique);
108 addBundledTrack("/resources/audio/startup3.mp3", unique);
109
110 if (playlist.isEmpty()) {
111 soundsEnabled = false;
112 }
113 }

190 private synchronized void playTrackAt(int index) {
191 if (playlist.isEmpty()) {
192 return;
193 }
194
195 if (!ensureJavafxReady()) {
196 soundsEnabled = false;
197 return;
198 }
199
200 int safeIndex = ((index % playlist.size()) + playlist.size()) % playlist.size();
201 URI uri = playlist.get(safeIndex);
202
203 disposeCurrentPlayer();
204
205 try {
206 Media media = new Media(uri.toString());
207 currentPlayer = new MediaPlayer(media);
208 currentPlayer.setVolume(masterVolume);
209 currentPlayer.setOnEndOfMedia(this::playNextTrack);
210 currentPlayer.setOnError(() -> {
211 System.err.println("Media player error: " + currentPlayer.getError());
212 playNextTrack();
213 });
214 currentTrackIndex = safeIndex;
215 currentPlayer.play();
216 } catch (Exception e) {
217 System.err.println("Cannot play track: " + uri + " -> " + e.getMessage());
218 playNextTrack();
219 }
220 }
