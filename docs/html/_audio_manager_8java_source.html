198 private void loadMusicPlaylist() {
199 try {
200 int trackNumber = 1;
201 boolean tracksFound = true;
202
203 System.out.println("Загрузка плейлиста фоновой музыки...");
204
205 while (tracksFound) {
206 String trackName = "/resources/audio/startup" + trackNumber + ".mp3";
207 URL trackUrl = getClass().getResource(trackName);
208
209 if (trackUrl != null) {
210 Media media = new Media(trackUrl.toString());
211 MediaPlayer player = new MediaPlayer(media);
212 player.setVolume(musicVolume);
213
214 // Настраиваем обработчик окончания трека
215 final int currentTrack = trackNumber - 1;
216 player.setOnEndOfMedia(() -> {
217 playNextTrack();
218 });
219
220 musicPlaylist.add(player);
221 System.out.println(" Загружен трек: " + trackName);
222 trackNumber++;
223 } else {
224 tracksFound = false;
225 }
226 }
227
228 // Проверяем, были ли загружены треки
229 if (musicPlaylist.isEmpty()) {
230 System.out.println("Треки не найдены, создаем заглушечный плейлист");
231 createDummyPlaylist();
232 } else {
233 System.out.println("Загружено треков: " + musicPlaylist.size());
234 }
235
236 } catch (Exception e) {
237 System.err.println("Ошибка загрузки плейлиста: " + e.getMessage());
238 createDummyPlaylist();
239 }

318 public void playNextTrack() {
319 if (musicPlaylist.isEmpty()) return;
320
321 // Останавливаем текущий трек
322 stopCurrentTrack();
323
324 // Выбираем следующий трек
325 if (shuffleMode) {
326 // В режиме перемешивания выбираем случайный трек
327 int newIndex;
328 do {
329 newIndex = random.nextInt(musicPlaylist.size());
330 } while (musicPlaylist.size() > 1 && newIndex == currentTrackIndex);
331
332 currentTrackIndex = newIndex;
333 } else {
334 // В обычном режиме переходим к следующему треку
335 currentTrackIndex++;
336
337 // Проверяем, не достигли ли конца плейлиста
338 if (currentTrackIndex >= musicPlaylist.size()) {
339 if (repeatMode) {
340 currentTrackIndex = 0; // Начинаем сначала
341 } else {
342 currentTrackIndex = musicPlaylist.size() - 1; // Останавливаемся на последнем
343 System.out.println("Конец плейлиста");
344 return;
345 }
346 }
347 }

349 playCurrentTrack();
350 System.out.println("Переключено на трек: " + (currentTrackIndex + 1) + "/" + musicPlaylist.size());
351 }

360 public void playPreviousTrack() {

362
363 stopCurrentTrack();
364
365 if (shuffleMode) {
366 // В режиме перемешивания выбираем случайный трек
367 int newIndex;
368 do {
369 newIndex = random.nextInt(musicPlaylist.size());
370 } while (musicPlaylist.size() > 1 && newIndex == currentTrackIndex);
371
372 currentTrackIndex = newIndex;

374 // В обычном режиме переходим к предыдущему треку
375 currentTrackIndex = (currentTrackIndex - 1 + musicPlaylist.size()) % musicPlaylist.size();
376 }
377
378 playCurrentTrack();
379 System.out.println("Переключено на трек: " + (currentTrackIndex + 1) + "/" + musicPlaylist.size());
