93 public static DatabaseConfig fromProperties(Properties properties, Path configFile, Path defaultSqlitePath) {
94 Objects.requireNonNull(properties, "properties");
95 Objects.requireNonNull(configFile, "configFile");
96 Objects.requireNonNull(defaultSqlitePath, "defaultSqlitePath");
97
98 Type type = parseType(read(properties, "db.type", "sqlite"));
99 Path sqlitePath = resolveSqlitePath(read(properties, "db.sqlitePath", ""), defaultSqlitePath);
100 String host = read(properties, "db.host", "");
101 int port = parsePort(read(properties, "db.port", ""), type == Type.POSTGRESQL ? 5432 : 0);
102 String databaseName = read(properties, "db.name", "");
103 String username = read(properties, "db.username", "");
104 String password = read(properties, "db.password", "");
105 String jdbcUrl = read(properties, "db.jdbcUrl", "");
106 String driverClass = read(properties, "db.driverClass", "");
107 String sslMode = read(properties, "db.sslMode", "require");
108
109 return new DatabaseConfig(
110 type,
111 configFile.toAbsolutePath(),
112 sqlitePath,
113 host,
114 port,
115 databaseName,
116 username,
117 password,
118 jdbcUrl,
119 driverClass,
120 sslMode
121 );
122 }

124 public void validate() {
125 switch (type) {
126 case SQLITE -> {
127 if (sqlitePath == null) {
128 throw new IllegalStateException("SQLite path is not configured");
129 }
130 }
131 case POSTGRESQL -> {
132 if (host.isBlank() || databaseName.isBlank() || username.isBlank()) {
133 throw new IllegalStateException(
134 "PostgreSQL config is incomplete. Fill db.host, db.name and db.username in " + configFile
135 );
136 }
137 }
138 case CUSTOM -> {
139 if (jdbcUrl.isBlank() || getDriverClass().isBlank()) {
140 throw new IllegalStateException(
141 "Custom JDBC config is incomplete. Fill db.jdbcUrl and db.driverClass in " + configFile
142 );
143 }
144 }
145 default -> throw new IllegalStateException("Unsupported DB type: " + type);
146 }
147 }

267 private static Type parseType(String value) {
268 String normalized = value.trim().toLowerCase();
269 return switch (normalized) {
270 case "", "sqlite" -> Type.SQLITE;
271 case "postgres", "postgresql" -> Type.POSTGRESQL;
272 case "custom", "jdbc" -> Type.CUSTOM;
273 default -> throw new IllegalArgumentException("Unsupported db.type: " + value);
274 };
275 }
276
277 private static int parsePort(String value, int defaultValue) {
278 if (value == null || value.isBlank()) {
279 return defaultValue;
280 }
281 return Integer.parseInt(value.trim());
282 }

284 private static Path resolveSqlitePath(String configuredPath, Path defaultPath) {
285 if (configuredPath == null || configuredPath.isBlank()) {
286 return defaultPath.toAbsolutePath();
287 }
288 Path path = Paths.get(configuredPath.trim());
289 return path.isAbsolute() ? path : path.toAbsolutePath();
290 }
