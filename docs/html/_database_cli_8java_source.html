63 private static void runProbe() throws Exception {
64 DatabaseConfig config = DatabaseConfig.load();
65 config.validate();
66 config.loadDriver();
67
68 if (config.isSqlite() && config.getSqlitePath() != null && config.getSqlitePath().getParent() != null) {
69 Files.createDirectories(config.getSqlitePath().getParent());
70 }
71
72 if (config.isCustom()) {
73 throw new IllegalStateException(
74 "Probe mode is implemented for SQLite/PostgreSQL only. For another DB engine, adapt SQL in DatabaseCli."
75 );
76 }
77
78 System.out.println("DB config: " + config.describe());
79
80 try (Connection connection = config.openConnection();
81 Statement statement = connection.createStatement()) {
82 statement.execute(createProbeTableSql(config));
83 }
84
85 String note = "probe from DatabaseCli at " + LocalDateTime.now().format(PROBE_NOTE_TIME);
86
87 try (Connection connection = config.openConnection();
88 PreparedStatement statement = connection.prepareStatement(
89 "INSERT INTO " + PROBE_TABLE + "(note) VALUES (?)"
90 )) {
91 statement.setString(1, note);
92 statement.executeUpdate();
93 }
94
95 try (Connection connection = config.openConnection();
96 Statement statement = connection.createStatement();
97 ResultSet rs = statement.executeQuery(
98 "SELECT id, note, created_at FROM " + PROBE_TABLE + " ORDER BY id DESC LIMIT 3"
99 )) {
100 System.out.println("Latest probe rows:");
101 while (rs.next()) {
102 System.out.printf(
103 " id=%s | note=%s | created_at=%s%n",
104 rs.getString("id"),
105 rs.getString("note"),
106 rs.getString("created_at")
107 );
108 }
109 }
110
111 try (Connection connection = config.openConnection();
112 Statement statement = connection.createStatement()) {
113 statement.execute("DROP TABLE " + PROBE_TABLE);
114 }
115
116 System.out.println("Probe completed successfully.");
117 }

119 private static void runSeed(String[] args) {
120 if (args.length < 2) {
121 throw new IllegalArgumentException("Missing task count. Example: seed 1000");
122 }
123
124 int count = Integer.parseInt(args[1]);
125 if (count <= 0) {
126 throw new IllegalArgumentException("Task count must be greater than 0");
127 }
128
129 DatabaseManager databaseManager = DatabaseManager.getInstance();
130 if (!databaseManager.isAvailable()) {
131 throw new IllegalStateException(
132 "Database is not available. Check db.properties and JDBC drivers before seeding."
133 );
134 }
135
136 System.out.println("Seeding " + count + " tasks into " + databaseManager.getConfig().describe());
137
138 LocalDateTime baseTime = LocalDateTime.now().withSecond(0).withNano(0);
139 for (int i = 0; i < count; i++) {
140 Task task = generateTask(i, baseTime);
141 databaseManager.saveTask(task);
142 databaseManager.addTagToTask(task.getId(), TAGS[i % TAGS.length]);
143 }
144
145 System.out.println("Created " + count + " tasks.");
146 }

148 private static Task generateTask(int index, LocalDateTime baseTime) {
149 Priority priority = Priority.values()[index % Priority.values().length];
150 Category category = Category.values()[index % Category.values().length];
151 LocalDateTime start = baseTime.plusHours(index % 48).plusDays(index / 48);
152 LocalDateTime end = start.plusMinutes(45 + (index % 6) * 15L);
153
154 Task task = new Task(
155 ACTIONS[index % ACTIONS.length] + " " + SUBJECTS[index % SUBJECTS.length] + " #" + (index + 1),
156 "Автоматически созданная задача №" + (index + 1) + " для проверки нагрузки и работы БД.",
157 start,
158 end,
159 priority,
160 category
161 );
162
163 task.setSortIndex(index + 1);
164 task.setReminderOffsetMinutes((index % 4) * 15);
165 if (index % 10 == 0) {
166 task.setRecurrenceRule("DAILY");
167 } else if (index % 15 == 0) {
168 task.setRecurrenceRule("WEEKLY");
169 }
170
171 return task;
172 }
