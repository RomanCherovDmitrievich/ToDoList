106 public void start() {
107 // Создаем случайное количество снежинок (80-100)
108 int snowflakeCount = 80 + random.nextInt(20);
109
110 // Создаем каждую снежинку и добавляем на панель
111 for (int i = 0; i < snowflakeCount; i++) {
112 Snowflake snowflake = createSnowflake();
113 snowflakes.add(snowflake);
114 snowPane.getChildren().add(snowflake.circle);
115 }
116
117 // Создаем и запускаем таймер анимации
118 animationTimer = new AnimationTimer() {
119 private long lastUpdate = 0;
120
130 @Override
131 public void handle(long now) {
132 // Первый вызов - инициализируем время
133 if (lastUpdate == 0) {
134 lastUpdate = now;
135 return;
136 }
137
138 // Вычисляем время, прошедшее с предыдущего кадра
139 double elapsedSeconds = (now - lastUpdate) / 1_000_000_000.0;
140
141 // Обновляем позиции всех снежинок
142 updateSnowflakes(elapsedSeconds);
143
144 // Сохраняем время текущего кадра
145 lastUpdate = now;
146 }
147 };
148
149 // Запускаем анимацию
150 animationTimer.start();
151 }

194 private Snowflake createSnowflake() {
195 Circle circle = new Circle();
196
197 // ОЧЕНЬ маленькие снежинки (1-3 пикселя)
198 double size = 1 + random.nextDouble() * 2;
199 circle.setRadius(size);
200
201 // Случайная начальная позиция (вверху экрана)
202 // Начинаем выше экрана для плавного появления
203 double x = random.nextDouble() * snowPane.getWidth();
204 double y = -size - random.nextDouble() * 100;
205
206 circle.setTranslateX(x);
207 circle.setTranslateY(y);
208
209 // Разная прозрачность (от легкой до более заметной)
210 double opacity = 0.3 + random.nextDouble() * 0.7;
211 circle.setOpacity(opacity);
212
213 // Разные оттенки белого/голубого для разнообразия
214 int colorChoice = random.nextInt(3);
215 switch (colorChoice) {
216 case 0:
217 circle.setFill(javafx.scene.paint.Color.WHITE); // Чистый белый
218 break;
219 case 1:
220 // AliceBlue - очень светлый голубой (#F0F8FF)
221 circle.setFill(javafx.scene.paint.Color.rgb(240, 248, 255));
222 break;
223 case 2:
224 // LightBlue - светлый голубой (#ADD8E6)
225 circle.setFill(javafx.scene.paint.Color.rgb(173, 216, 230));
226 break;
227 }
228
229 // Добавляем легкое свечение для эффекта "сияния"
230 circle.setEffect(new javafx.scene.effect.Glow(0.3));
231
232 // Разная скорость падения (от медленной до быстрой)
233 double speed = 30 + random.nextDouble() * 70; // пикселей в секунду
234
235 // Параметры для покачивания из стороны в сторону
236 double sway = random.nextDouble() * 2 - 1; // начальное смещение в диапазоне [-1, 1]
237 double swaySpeed = 0.5 + random.nextDouble() * 2; // скорость покачивания
238
239 return new Snowflake(circle, speed, sway, swaySpeed);
240 }

260 private void updateSnowflakes(double deltaTime) {
261 double paneHeight = snowPane.getHeight();
262 double paneWidth = snowPane.getWidth();
263 long time = System.currentTimeMillis(); // Текущее время для анимации
264
265 for (Snowflake snowflake : snowflakes) {
266 Circle circle = snowflake.circle;
267
268 // Двигаем снежинку вниз с её скоростью
269 double newY = circle.getTranslateY() + snowflake.speed * deltaTime;
270
271 // Покачивание из стороны в сторону по синусоидальному закону
272 double currentX = circle.getTranslateX();
273 double swayOffset = Math.sin(time * 0.001 * snowflake.swaySpeed + snowflake.sway) * 15;
274 double newX = currentX + swayOffset * deltaTime;
275
276 // Если снежинка упала за нижнюю границу, перемещаем наверх
277 if (newY > paneHeight + circle.getRadius()) {
278 newY = -circle.getRadius(); // Начинаем выше экрана
279 newX = random.nextDouble() * paneWidth; // Случайная позиция по X
280
281 // Немного меняем параметры при перерождении для разнообразия
282 snowflake.speed = 30 + random.nextDouble() * 70;
283 snowflake.swaySpeed = 0.5 + random.nextDouble() * 2;
284 }
285
286 // Если снежинка вышла за боковые границы, возвращаем её
287 if (newX < -circle.getRadius()) {
288 newX = paneWidth + circle.getRadius(); // Появляется справа
289 } else if (newX > paneWidth + circle.getRadius()) {
290 newX = -circle.getRadius(); // Появляется слева
291 }
292
293 // Применяем новые координаты
294 circle.setTranslateX(newX);
295 circle.setTranslateY(newY);
296
297 // Легкое мерцание для реалистичности
298 double flicker = 0.7 + 0.3 * Math.sin(time * 0.003 + snowflake.sway);
299 circle.setOpacity(flicker * (0.3 + random.nextDouble() * 0.4));
300 }
301 }
