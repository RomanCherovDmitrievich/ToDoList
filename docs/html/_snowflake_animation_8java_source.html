110 public void start() {
111 // Создаем случайное количество снежинок (80-100)
112 int snowflakeCount = 80 + random.nextInt(20);
113
114 // Создаем каждую снежинку и добавляем на панель
115 for (int i = 0; i < snowflakeCount; i++) {
116 Snowflake snowflake = createSnowflake();
117 snowflakes.add(snowflake);
118 snowPane.getChildren().add(snowflake.circle);
119 }
120
121 // Создаем и запускаем таймер анимации
122 animationTimer = new AnimationTimer() {
123 private long lastUpdate = 0;
124
134 @Override
135 public void handle(long now) {
136 // Первый вызов - инициализируем время
137 if (lastUpdate == 0) {
138 lastUpdate = now;
139 return;
140 }
141
142 // Вычисляем время, прошедшее с предыдущего кадра
143 double elapsedSeconds = (now - lastUpdate) / 1_000_000_000.0;
144
145 // Обновляем позиции всех снежинок
146 updateSnowflakes(elapsedSeconds);
147
148 // Сохраняем время текущего кадра
149 lastUpdate = now;
150 }
151 };
152
153 // Запускаем анимацию
154 animationTimer.start();
155 }

198 private Snowflake createSnowflake() {
199 Circle circle = new Circle();
200
201 // ОЧЕНЬ маленькие снежинки (1-3 пикселя)
202 double size = 1 + random.nextDouble() * 2;
203 circle.setRadius(size);
204
205 // Случайная начальная позиция (вверху экрана)
206 // Начинаем выше экрана для плавного появления
207 double x = random.nextDouble() * snowPane.getWidth();
208 double y = -size - random.nextDouble() * 100;
209
210 circle.setTranslateX(x);
211 circle.setTranslateY(y);
212
213 // Разная прозрачность (от легкой до более заметной)
214 double opacity = 0.3 + random.nextDouble() * 0.7;
215 circle.setOpacity(opacity);
216
217 // Разные оттенки белого/голубого для разнообразия
218 int colorChoice = random.nextInt(3);
219 switch (colorChoice) {
220 case 0:
221 circle.setFill(javafx.scene.paint.Color.WHITE); // Чистый белый
222 break;
223 case 1:
224 // AliceBlue - очень светлый голубой (#F0F8FF)
225 circle.setFill(javafx.scene.paint.Color.rgb(240, 248, 255));
226 break;
227 case 2:
228 // LightBlue - светлый голубой (#ADD8E6)
229 circle.setFill(javafx.scene.paint.Color.rgb(173, 216, 230));
230 break;
231 }
232
233 // Добавляем легкое свечение для эффекта "сияния"
234 circle.setEffect(new javafx.scene.effect.Glow(0.3));
235
236 // Разная скорость падения (от медленной до быстрой)
237 double speed = 30 + random.nextDouble() * 70; // пикселей в секунду
238
239 // Параметры для покачивания из стороны в сторону
240 double sway = random.nextDouble() * 2 - 1; // начальное смещение в диапазоне [-1, 1]
241 double swaySpeed = 0.5 + random.nextDouble() * 2; // скорость покачивания
242
243 return new Snowflake(circle, speed, sway, swaySpeed);
244 }

264 private void updateSnowflakes(double deltaTime) {
265 double paneHeight = snowPane.getHeight();
266 double paneWidth = snowPane.getWidth();
267 long time = System.currentTimeMillis(); // Текущее время для анимации
268
269 for (Snowflake snowflake : snowflakes) {
270 Circle circle = snowflake.circle;
271
272 // Двигаем снежинку вниз с её скоростью
273 double newY = circle.getTranslateY() + snowflake.speed * deltaTime;
274
275 // Покачивание из стороны в сторону по синусоидальному закону
276 double currentX = circle.getTranslateX();
277 double swayOffset = Math.sin(time * 0.001 * snowflake.swaySpeed + snowflake.sway) * 15;
278 double newX = currentX + swayOffset * deltaTime;
279
280 // Если снежинка упала за нижнюю границу, перемещаем наверх
281 if (newY > paneHeight + circle.getRadius()) {
282 newY = -circle.getRadius(); // Начинаем выше экрана
283 newX = random.nextDouble() * paneWidth; // Случайная позиция по X
284
285 // Немного меняем параметры при перерождении для разнообразия
286 snowflake.speed = 30 + random.nextDouble() * 70;
287 snowflake.swaySpeed = 0.5 + random.nextDouble() * 2;
288 }
289
290 // Если снежинка вышла за боковые границы, возвращаем её
291 if (newX < -circle.getRadius()) {
292 newX = paneWidth + circle.getRadius(); // Появляется справа
293 } else if (newX > paneWidth + circle.getRadius()) {
294 newX = -circle.getRadius(); // Появляется слева
295 }
296
297 // Применяем новые координаты
298 circle.setTranslateX(newX);
299 circle.setTranslateY(newY);
300
301 // Легкое мерцание для реалистичности
302 double flicker = 0.7 + 0.3 * Math.sin(time * 0.003 + snowflake.sway);
303 circle.setOpacity(flicker * (0.3 + random.nextDouble() * 0.4));
304 }
305 }
