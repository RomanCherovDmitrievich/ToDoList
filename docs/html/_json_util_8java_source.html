372 private static List<Task> parseTasksFromJson(String jsonContent) {
373 List<Task> tasks = new ArrayList<>();
374
375 if (jsonContent == null || jsonContent.trim().isEmpty()) {
376 return tasks;
377 }
378
379 String cleanJson = jsonContent.trim();
380
381 // Проверяем, что JSON начинается с '[' и заканчивается ']'
382 if (!cleanJson.startsWith("[") || !cleanJson.endsWith("]")) {
383 // Возможно это один объект, а не массив
384 if (cleanJson.startsWith("{") && cleanJson.endsWith("}")) {
385 cleanJson = "[" + cleanJson + "]";
386 } else {
387 throw new IllegalArgumentException("Неверный формат JSON. Ожидается массив или объект.");
388 }
389 }
390
391 // Убираем внешние скобки
392 String arrayContent = cleanJson.substring(1, cleanJson.length() - 1).trim();
393
394 if (arrayContent.isEmpty()) {
395 return tasks;
396 }
397
398 // Разделяем объекты в массиве
399 List<String> taskStrings = splitJsonObjects(arrayContent);
400
401 for (String taskJson : taskStrings) {
402 try {
403 Task task = parseSingleTask(taskJson.trim());
404 if (task != null) {
405 tasks.add(task);
406 }
407 } catch (Exception e) {
408 System.err.println("Ошибка парсинга задачи: " + e.getMessage());
409 // Продолжаем с другими задачами
410 }
411 }
412
413 return tasks;
414 }

426 private static List<String> splitJsonObjects(String arrayContent) {
427 List<String> objects = new ArrayList<>();
428 StringBuilder current = new StringBuilder();
429 int braceCount = 0;
430 boolean inQuotes = false;
431 boolean escapeNext = false;
432
433 for (int i = 0; i < arrayContent.length(); i++) {
434 char c = arrayContent.charAt(i);
435
436 if (escapeNext) {
437 escapeNext = false;
438 current.append(c);
439 continue;
440 }
441
442 if (c == '\\') {
443 escapeNext = true;
444 current.append(c);
445 continue;
446 }
447
448 if (c == '"' && !escapeNext) {
449 inQuotes = !inQuotes;
450 }
451
452 if (!inQuotes) {
453 if (c == '{') {
454 braceCount++;
455 } else if (c == '}') {
456 braceCount--;
457 }
458 }
459
460 current.append(c);
461
462 if (!inQuotes && braceCount == 0 && c == '}') {
463 // Нашли полный объект
464 objects.add(current.toString());
465 current.setLength(0);
466
467 // Пропускаем запятую после объекта, если есть
468 if (i + 1 < arrayContent.length() && arrayContent.charAt(i + 1) == ',') {
469 i++;
470 }
471 }
472 }
473
474 return objects;
475 }

492 private static Task parseSingleTask(String json) {
493 if (json == null || json.trim().isEmpty()) {
494 return null;
495 }
496
497 String content = json.trim();
498
499 // Убираем окружающие фигурные скобки
500 if (content.startsWith("{") && content.endsWith("}")) {
501 content = content.substring(1, content.length() - 1);
502 }
503
504 // Разбиваем на пары ключ-значение
505 Map<String, String> fields = parseJsonFields(content);
506
507 try {
508 Task task = new Task(
509 fields.getOrDefault("id", UUID.randomUUID().toString()),
510 fields.getOrDefault("title", ""),
511 fields.getOrDefault("description", ""),
512 parseDateTime(fields.get("startTime")),
513 parseDateTime(fields.get("endTime")),
514 fields.getOrDefault("priority", "IMPORTANT"),
515 fields.getOrDefault("category", "OTHER"),
516 Boolean.parseBoolean(fields.getOrDefault("completed", "false")),
517 Boolean.parseBoolean(fields.getOrDefault("overdue", "false")),
518 parseDateTime(fields.get("createdAt"))
519 );
520 String sortIndex = fields.getOrDefault("sortIndex", "0");
521 String recurrenceRule = fields.getOrDefault("recurrenceRule", "NONE");
522 String reminderOffset = fields.getOrDefault("reminderOffsetMinutes", "0");
523 String recurrenceEnd = fields.getOrDefault("recurrenceEnd", "");
524
525 try {
526 task.setSortIndex(Integer.parseInt(sortIndex));
527 } catch (NumberFormatException ignored) {
528 }
529 task.setRecurrenceRule(recurrenceRule);
530 try {
531 task.setReminderOffsetMinutes(Integer.parseInt(reminderOffset));
532 } catch (NumberFormatException ignored) {
533 }
534 if (!recurrenceEnd.isBlank()) {
535 task.setRecurrenceEnd(LocalDateTime.parse(parseDateTime(recurrenceEnd), DateTimeFormatter.ISO_LOCAL_DATE_TIME));
536 }
537
538 return task;
539 } catch (Exception e) {
540 System.err.println("Ошибка создания задачи из JSON: " + e.getMessage());
541 throw new RuntimeException("Не удалось создать задачу из JSON", e);

555 private static Map<String, String> parseJsonFields(String jsonContent) {
556 Map<String, String> fields = new HashMap<>();
557
558 StringBuilder currentKey = new StringBuilder();
559 StringBuilder currentValue = new StringBuilder();
560 boolean inKey = true;
561 boolean inQuotes = false;
562 boolean escapeNext = false;
563 int braceCount = 0;
564 int bracketCount = 0;
565
566 for (int i = 0; i < jsonContent.length(); i++) {
567 char c = jsonContent.charAt(i);
568
569 if (escapeNext) {
570 escapeNext = false;
571 if (inKey) {
572 currentKey.append(c);
573 } else {
574 currentValue.append(c);
575 }
576 continue;
577 }
578
579 if (c == '\\') {
580 escapeNext = true;
581 if (inKey) {
582 currentKey.append(c);
583 } else {
584 currentValue.append(c);
585 }
586 continue;
587 }
588
589 if (c == '"' && !escapeNext) {
590 inQuotes = !inQuotes;
591 }
592
593 if (!inQuotes) {
594 if (c == '{') braceCount++;
595 if (c == '}') braceCount--;
596 if (c == '[') bracketCount++;
597 if (c == ']') bracketCount--;
598 }
599
600 if (!inQuotes && c == ':' && braceCount == 0 && bracketCount == 0 && inKey) {
601 inKey = false;
602 continue;
603 }
604
605 if (!inQuotes && c == ',' && braceCount == 0 && bracketCount == 0 && !inKey) {
606 // Сохраняем пару ключ-значение
607 String key = currentKey.toString().trim().replace("\"", "");
608 String value = currentValue.toString().trim();
609
610 // Убираем кавычки из значения, если они есть
611 if (value.startsWith("\"") && value.endsWith("\"")) {
612 value = value.substring(1, value.length() - 1);
613 }
614
615 fields.put(key, value);
616
617 // Сбрасываем для следующей пары
618 currentKey.setLength(0);
619 currentValue.setLength(0);
620 inKey = true;
621 continue;
622 }
623
624 if (inKey) {
625 currentKey.append(c);
626 } else {
627 currentValue.append(c);
628 }
629 }
630
631 // Добавляем последнюю пару
632 if (currentKey.length() > 0) {
633 String key = currentKey.toString().trim().replace("\"", "");
634 String value = currentValue.toString().trim();
635
636 if (value.startsWith("\"") && value.endsWith("\"")) {
637 value = value.substring(1, value.length() - 1);
638 }
639
640 fields.put(key, value);
641 }
642
643 return fields;
644 }

657 private static String parseDateTime(String dateTimeStr) {
658 if (dateTimeStr == null || dateTimeStr.isEmpty()) {
659 return LocalDateTime.now().format(DateTimeFormatter.ISO_LOCAL_DATE_TIME);
660 }
661
662 try {
663 // Пробуем парсить с нашим форматером
664 LocalDateTime.parse(dateTimeStr, FORMATTER);
665 return dateTimeStr;
666 } catch (Exception e1) {
667 try {
668 // Пробуем стандартный ISO формат
669 LocalDateTime.parse(dateTimeStr, DateTimeFormatter.ISO_LOCAL_DATE_TIME);
670 return dateTimeStr;
671 } catch (Exception e2) {
672 try {
673 // Пробуем без времени
674 LocalDateTime dateTime = LocalDateTime.parse(dateTimeStr + "T00:00:00");
675 return dateTime.format(DateTimeFormatter.ISO_LOCAL_DATE_TIME);
676 } catch (Exception e3) {
677 System.err.println("Не удалось распарсить дату: " + dateTimeStr);
678 return LocalDateTime.now().format(DateTimeFormatter.ISO_LOCAL_DATE_TIME);
679 }
680 }
