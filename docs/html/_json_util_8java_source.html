301 private static List<Task> parseTasksFromJson(String jsonContent) {
302 List<Task> tasks = new ArrayList<>();
303
304 // Убираем пробелы и переносы строк
305 String cleanJson = jsonContent.trim();
306
307 // Проверяем, что JSON начинается с '[' и заканчивается ']'
308 if (!cleanJson.startsWith("[") || !cleanJson.endsWith("]")) {
309 throw new IllegalArgumentException("Неверный формат JSON. Ожидается массив.");
310 }
311
312 // Убираем внешние скобки
313 String arrayContent = cleanJson.substring(1, cleanJson.length() - 1).trim();
314
315 if (arrayContent.isEmpty()) {
316 return tasks;
317 }
318
319 // Разделяем объекты в массиве
320 List<String> taskStrings = splitJsonObjects(arrayContent);
321
322 for (String taskJson : taskStrings) {
323 try {
324 Task task = parseSingleTask(taskJson.trim());
325 if (task != null) {
326 tasks.add(task);
327 }
328 } catch (Exception e) {
329 System.err.println("Ошибка парсинга задачи: " + e.getMessage());
330 e.printStackTrace();
331 }
332 }
333
334 return tasks;

401 private static Task parseSingleTask(String json) {
402 // Убираем окружающие фигурные скобки
403 String content = json.substring(1, json.length() - 1);
404
405 // Разбиваем на пары ключ-значение
406 Map<String, String> fields = new HashMap<>();
407
408 StringBuilder currentKey = new StringBuilder();
409 StringBuilder currentValue = new StringBuilder();
410 boolean inKey = true;
411 boolean inQuotes = false;
412 int braceCount = 0;
413 int bracketCount = 0;
414
415 for (int i = 0; i < content.length(); i++) {
416 char c = content.charAt(i);
417
418 if (c == '"' && (i == 0 || content.charAt(i - 1) != '\\')) {
419 inQuotes = !inQuotes;
420 }
421
422 if (!inQuotes) {
423 if (c == '{') braceCount++;
424 if (c == '}') braceCount--;
425 if (c == '[') bracketCount++;
426 if (c == ']') bracketCount--;
427 }
428
429 if (!inQuotes && c == ':' && braceCount == 0 && bracketCount == 0) {
430 inKey = false;
431 continue;
432 }
433
434 if (!inQuotes && c == ',' && braceCount == 0 && bracketCount == 0) {
435 // Сохраняем пару ключ-значение
436 String key = currentKey.toString().trim().replace("\"", "");
437 String value = currentValue.toString().trim();
438
439 if (value.startsWith("\"") && value.endsWith("\"")) {
440 value = value.substring(1, value.length() - 1);
441 }
442
443 fields.put(key, value);
444
445 // Сбрасываем для следующей пары
446 currentKey.setLength(0);
447 currentValue.setLength(0);
448 inKey = true;
449 continue;
450 }
451
452 if (inKey) {
453 currentKey.append(c);
454 } else {
455 currentValue.append(c);
456 }
457 }
458
459 // Добавляем последнюю пару
460 if (currentKey.length() > 0) {
461 String key = currentKey.toString().trim().replace("\"", "");
462 String value = currentValue.toString().trim();
463
464 if (value.startsWith("\"") && value.endsWith("\"")) {
465 value = value.substring(1, value.length() - 1);
466 }
467
468 fields.put(key, value);
469 }
470
471 // Создаем объект Task
472 try {
473 return new Task(
474 fields.get("id"),
475 fields.get("title"),
476 fields.get("description"),
477 fields.get("startTime"),
478 fields.get("endTime"),
479 fields.get("priority"),
480 fields.get("category"),
481 Boolean.parseBoolean(fields.get("completed")),
482 Boolean.parseBoolean(fields.get("overdue")),
483 fields.get("createdAt")
484 );
485 } catch (Exception e) {
486 System.err.println("Ошибка создания задачи из JSON: " + e.getMessage());
487 throw new RuntimeException("Не удалось создать задачу из JSON", e);
488 }
489 }
