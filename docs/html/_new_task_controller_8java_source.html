225 private boolean validateForm() {
226 boolean isValid = true;
227 StringBuilder errorMessage = new StringBuilder();
228
229 // Проверка заголовка
230 if (titleField.getText() == null || titleField.getText().trim().isEmpty()) {
231 isValid = false;
232 errorMessage.append("• Заголовок не может быть пустым\n");
233 titleField.setStyle("-fx-border-color: red;");
234 } else {
235 titleField.setStyle("");
236 }
237
238 // Проверка дат
239 if (startDatePicker.getValue() == null || endDatePicker.getValue() == null) {
240 isValid = false;
241 errorMessage.append("• Укажите даты начала и окончания\n");
242 } else if (endDatePicker.getValue().isBefore(startDatePicker.getValue())) {
243 isValid = false;
244 errorMessage.append("• Дата окончания не может быть раньше даты начала\n");
245 endDatePicker.setStyle("-fx-border-color: red;");
246 } else {
247 endDatePicker.setStyle("");
248 }
249
250 // Обновляем сообщение об ошибке
251 if (!isValid) {
252 errorLabel.setText(errorMessage.toString());
253 errorLabel.setTextFill(Color.RED);
254 } else {
255 errorLabel.setText("");
256 }
257
258 // Активируем/деактивируем кнопку создания
259 createButton.setDisable(!isValid);
260
261 return isValid;
262 }

311 );
312 LocalDateTime endDateTime = LocalDateTime.of(endDate, endTime);
313
314 // Получаем приоритет и категорию
315 Priority priority = Priority.fromDisplayName(priorityCombo.getValue());
316 Category category = Category.fromDisplayName(categoryCombo.getValue());
317
318 // Создаем новую задачу
319 createdTask = new Task(title, description, startDateTime, endDateTime, priority, category);
320 taskCreated = true;
321

324 stage.close();
325
326 } catch (Exception e) {

328 errorLabel.setTextFill(Color.RED);
