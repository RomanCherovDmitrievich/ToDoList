149 private void applyTaskToForm(Task task) {
150 dialogTitleLabel.setText("Редактировать задачу");
151 createButton.setText("Сохранить");
152
153 titleField.setText(task.getTitle());
154 descriptionArea.setText(task.getDescription());
155
156 startDatePicker.setValue(task.getStartTime().toLocalDate());
157 endDatePicker.setValue(task.getEndTime().toLocalDate());
158
159 startHourCombo.setValue(String.format("%02d", task.getStartTime().getHour()));
160 startMinuteCombo.setValue(String.format("%02d", task.getStartTime().getMinute()));
161 endHourCombo.setValue(String.format("%02d", task.getEndTime().getHour()));
162 endMinuteCombo.setValue(String.format("%02d", task.getEndTime().getMinute()));
163
164 priorityCombo.setValue(task.getPriority().getDisplayName());
165 categoryCombo.setValue(task.getCategory().getDisplayName());
166
167 int reminderMinutes = task.getReminderOffsetMinutes();
168 if (reminderMinutes <= 0) {
169 reminderValueSpinner.getValueFactory().setValue(0);
170 reminderUnitChoice.setValue("мин");
171 } else if (reminderMinutes % (24 * 60) == 0) {
172 reminderValueSpinner.getValueFactory().setValue(reminderMinutes / (24 * 60));
173 reminderUnitChoice.setValue("день");
174 } else if (reminderMinutes % 60 == 0) {
175 reminderValueSpinner.getValueFactory().setValue(reminderMinutes / 60);
176 reminderUnitChoice.setValue("час");
177 } else {
178 reminderValueSpinner.getValueFactory().setValue(reminderMinutes);
179 reminderUnitChoice.setValue("мин");
180 }
181
182 String rule = task.getRecurrenceRule();
183 if (rule == null || rule.isBlank() || "NONE".equalsIgnoreCase(rule)) {
184 recurrenceCombo.setValue("Не повторять");
185 recurrenceRuleField.setText("");
186 } else if ("DAILY".equalsIgnoreCase(rule)) {
187 recurrenceCombo.setValue("Каждый день");
188 } else if ("WEEKLY".equalsIgnoreCase(rule)) {
189 recurrenceCombo.setValue("Каждую неделю");
190 } else if ("WEEKDAYS".equalsIgnoreCase(rule)) {
191 recurrenceCombo.setValue("Каждый будний день");
192 } else if ("MONTHLY".equalsIgnoreCase(rule)) {
193 recurrenceCombo.setValue("Каждый месяц");
194 } else {
195 recurrenceCombo.setValue("Пользовательское правило");
196 recurrenceRuleField.setText(rule);
197 }
198
199 if (task.getRecurrenceEnd() != null) {
200 recurrenceEndPicker.setValue(task.getRecurrenceEnd().toLocalDate());
201 }
202
203 updateRecurrenceFields();
204 }

206 private boolean validateForm() {
207 boolean isValid = true;
208 StringBuilder errorMessage = new StringBuilder();
209
210 if (titleField.getText() == null || titleField.getText().trim().isEmpty()) {
211 isValid = false;
212 errorMessage.append("• Заголовок не может быть пустым\n");
213 titleField.setStyle("-fx-border-color: red;");
214 } else {
215 titleField.setStyle("");
216 }
217
218 if (startDatePicker.getValue() == null || endDatePicker.getValue() == null) {
219 isValid = false;
220 errorMessage.append("• Укажите даты начала и окончания\n");
221 } else if (endDatePicker.getValue().isBefore(startDatePicker.getValue())) {
222 isValid = false;
223 errorMessage.append("• Дата окончания не может быть раньше даты начала\n");
224 endDatePicker.setStyle("-fx-border-color: red;");
225 } else {
226 endDatePicker.setStyle("");
227 }
228
229 if ("Пользовательское правило".equals(recurrenceCombo.getValue())) {
230 String rule = normalizeCustomRule(recurrenceRuleField.getText());
231 if (rule.isBlank()) {
232 isValid = false;
233 errorMessage.append("• Укажите RRULE для повторения\n");
234 } else if (RecurrenceUtil.nextOccurrence(LocalDateTime.now(), rule) == null) {
235 isValid = false;
236 errorMessage.append("• RRULE не распознано. Проверьте формат\n");
237 }
238 }
239
240 if (!isValid) {
241 errorLabel.setText(errorMessage.toString());
242 errorLabel.setTextFill(Color.RED);
243 } else {
244 errorLabel.setText("");
245 }
246
247 createButton.setDisable(!isValid);
248 return isValid;
249 }

327 private int toMinutes(Integer value, String unit) {
328 int amount = value == null ? 0 : value;
329 if (amount <= 0) {
330 return 0;
331 }
332 if ("час".equals(unit)) {
333 return amount * 60;
334 }
335 if ("день".equals(unit)) {
336 return amount * 24 * 60;
337 }
338 return amount;
339 }

341 private String buildRecurrenceRule() {
342 String selected = recurrenceCombo.getValue();
343 if (selected == null || "Не повторять".equals(selected)) {
344 return "NONE";
345 }
346 return switch (selected) {
347 case "Каждый день" -> "DAILY";
348 case "Каждую неделю" -> "WEEKLY";
349 case "Каждый будний день" -> "WEEKDAYS";
350 case "Каждый месяц" -> "MONTHLY";
351 case "Пользовательское правило" -> normalizeCustomRule(recurrenceRuleField.getText());
352 default -> "NONE";
353 };
354 }
355
356 private String normalizeCustomRule(String rawRule) {
357 if (rawRule == null) {
358 return "";
359 }
360 String rule = rawRule.trim();
361 if (rule.isBlank()) {
362 return "";
363 }
364 if (!rule.toUpperCase().startsWith("RRULE")) {
365 if (rule.toUpperCase().contains("FREQ=")) {
366 rule = "RRULE:" + rule;
367 }
368 }
369 return rule;
370 }
