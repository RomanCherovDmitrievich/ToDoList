38 private static LocalDateTime parseRRule(LocalDateTime start, String rule) {

91 private static LocalDateTime nextWeekly(LocalDateTime start, int interval, List<DayOfWeek> byDays) {
92 if (byDays.isEmpty()) {
93 return start.plusWeeks(interval);
94 }
95
96 LocalDateTime candidate = start.plusDays(1);
97 while (true) {
98 if (byDays.contains(candidate.getDayOfWeek()) && !candidate.isBefore(start.plusDays(1))) {
99 return candidate;
100 }
101 candidate = candidate.plusDays(1);
102 if (candidate.isAfter(start.plusWeeks(interval))) {
103 return candidate;
104 }
105 }
106 }

108 private static LocalDateTime nextMonthly(LocalDateTime start, int interval, List<DayOfWeek> byDays, int bySetPos) {
109 LocalDate base = start.toLocalDate().plusMonths(interval);
110
111 if (byDays.isEmpty()) {
112 return LocalDateTime.of(base.withDayOfMonth(Math.min(start.getDayOfMonth(), base.lengthOfMonth())), start.toLocalTime());
113 }
114
115 DayOfWeek target = byDays.get(0);
116 if (bySetPos == 0) {
117 bySetPos = 1;
118 }
119
120 LocalDate first = base.with(TemporalAdjusters.firstInMonth(target));
121 LocalDate selected = first.plusWeeks(bySetPos - 1);
122 if (selected.getMonth() != base.getMonth()) {
123 selected = base.with(TemporalAdjusters.lastInMonth(target));
124 }
125
126 return LocalDateTime.of(selected, start.toLocalTime());
127 }
