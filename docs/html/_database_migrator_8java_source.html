185 public static String synchronizeData() {
186 try {
187 StringBuilder result = new StringBuilder();
188 result.append("Синхронизация данных:\n");
189
190 // Загружаем из JSON
191 List<Task> jsonTasks = JsonUtil.loadTasks();
192 result.append("Задач в JSON: ").append(jsonTasks.size()).append("\n");
193
194 // Загружаем из БД
195 DatabaseManager dbManager = DatabaseManager.getInstance();
196 List<Task> dbTasks = dbManager.getAllTasks();
197 result.append("Задач в БД: ").append(dbTasks.size()).append("\n");
198
199 // Находим различия
200 int addedToDB = 0;
201 int addedToJSON = 0;
202
203 // Добавляем в БД задачи, которых там нет
204 for (Task jsonTask : jsonTasks) {
205 boolean existsInDB = dbTasks.stream()
206 .anyMatch(dbTask -> dbTask.getId().equals(jsonTask.getId()));
207 if (!existsInDB) {
208 dbManager.saveTask(jsonTask);
209 addedToDB++;
210 }
211 }
212
213 // Добавляем в JSON задачи, которых там нет
214 for (Task dbTask : dbTasks) {
215 boolean existsInJSON = jsonTasks.stream()
216 .anyMatch(jsonTask -> jsonTask.getId().equals(dbTask.getId()));
217 if (!existsInJSON) {
218 jsonTasks.add(dbTask);
219 addedToJSON++;
220 }
221 }
222
223 // Сохраняем обновленный JSON
224 if (addedToJSON > 0) {
225 JsonUtil.saveTasks(jsonTasks);
226 }
227
228 result.append("Добавлено в БД: ").append(addedToDB).append("\n");
229 result.append("Добавлено в JSON: ").append(addedToJSON).append("\n");
230 result.append("Синхронизация завершена!");
231
232 return result.toString();
233
234 } catch (Exception e) {
235 return "Ошибка синхронизации: " + e.getMessage();
236 }
