69 private void checkTasks() {
70 List<Task> tasks = TaskManager.getInstance().getAllTasks();
71 NotificationSettings settings = NotificationSettings.load();
72 LocalDateTime now = LocalDateTime.now();
73
74 Set<String> activeIds = new HashSet<>();
75
76 for (Task task : tasks) {
77 String id = task.getId();
78 activeIds.add(id);
79
80 if (task.isCompleted()) {
81 upcomingNotified.remove(id);
82 overdueNotified.remove(id);
83 continue;
84 }
85
86 int offset = task.getReminderOffsetMinutes() > 0 ? task.getReminderOffsetMinutes() : settings.getReminderOffsetMinutes();
87 LocalDateTime start = task.getStartTime();
88 LocalDateTime upcomingEdge = now.plusMinutes(Math.max(1, offset));
89
90 if (!start.isBefore(now) && !start.isAfter(upcomingEdge)) {
91 long minutes = Math.max(0, Duration.between(now, start).toMinutes());
92 if (upcomingNotified.add(id)) {
93 notifyAllChannels(settings, "Скоро начнется задача",
94 task.getTitle() + " (через " + minutes + " мин.)");
95 }
96 }
97
98 LocalDateTime end = task.getEndTime();
99 if (now.isAfter(end)) {
100 if (overdueNotified.add(id)) {
101 notifyAllChannels(settings, "Задача просрочена",
102 task.getTitle() + " — дедлайн был " + task.getFormattedEndTime());
103 }
104 } else {
105 overdueNotified.remove(id);
106 }
107 }
108
109 upcomingNotified.retainAll(activeIds);
110 overdueNotified.retainAll(activeIds);
111 }
