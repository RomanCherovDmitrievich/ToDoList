Создает приоритет из читаемого имени

Создает приоритет из читаемого имени Преобразует строковое имя приоритета в соответствующее enum значение

Аргументы
nameЧитаемое имя приоритета (регистронезависимое)
Возвращает
Priority Соответствующее enum значение или IMPORTANT, если имя не найдено
Исключения
NullPointerExceptionесли name равен null
Заметки
Поиск выполняется по нижнему регистру имени
Если имя не найдено, возвращается приоритет IMPORTANT (по умолчанию)

Priority p = Priority.fromDisplayName("Срочно"); // возвращает Priority.URGENT Priority p2 = Priority.fromDisplayName("срочно"); // также возвращает Priority.URGENT Priority p3 = Priority.fromDisplayName("Неизвестно"); // возвращает Priority.IMPORTANT

package model;
import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.Map;
public enum Priority {
URGENT("Срочно", "#FF4444", "urgent"),
// Это эквивалентно вызову: new Priority("Срочно", "#FF4444", "urgent");
IMPORTANT("Важно", "#FFBB33", "important"),
NORMAL("Желательно", "#00C851", "normal");
private final String displayName;
private final String color;
private final String iconName;
private static final Map<String, Priority> BY_NAME = new HashMap<>();
static {
for (Priority p : values()) {
BY_NAME.put(p.displayName.toLowerCase(), p);
}
}
Priority(String displayName, String color, String iconName) {
this.displayName = displayName;
this.color = color;
this.iconName = iconName;
}
public String getDisplayName() {
return displayName;
}
public String getColor() {
return color;
}
public String getIconName() {
return iconName;
}
public static Priority fromDisplayName(String name) {
return BY_NAME.getOrDefault(name.toLowerCase(), IMPORTANT);
}
public static Priority getDefault() {
return IMPORTANT;
}
}
