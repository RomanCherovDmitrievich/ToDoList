405 private void createNextRecurringTask(Task task) {
406 String rule = task.getRecurrenceRule();
407 if (rule == null || rule.isBlank() || rule.equalsIgnoreCase("NONE")) {
408 return;
409 }
410
411 LocalDateTime nextStart = RecurrenceUtil.nextOccurrence(task.getStartTime(), rule);
412 if (nextStart == null) {
413 return;
414 }
415
416 if (task.getRecurrenceEnd() != null && nextStart.isAfter(task.getRecurrenceEnd())) {
417 return;
418 }
419
420 long durationMinutes = java.time.Duration.between(task.getStartTime(), task.getEndTime()).toMinutes();
421 if (durationMinutes < 0) {
422 durationMinutes = 60;
423 }
424
425 LocalDateTime nextEnd = nextStart.plusMinutes(durationMinutes);
426
427 Task next = new Task(
428 task.getTitle(),
429 task.getDescription(),
430 nextStart,
431 nextEnd,
432 task.getPriority(),
433 task.getCategory()
434 );
435 next.setRecurrenceRule(rule);
436 next.setReminderOffsetMinutes(task.getReminderOffsetMinutes());
437 next.setRecurrenceEnd(task.getRecurrenceEnd());
438 next.setSortIndex(nextSortIndex());
439 addTask(next);
440 }
